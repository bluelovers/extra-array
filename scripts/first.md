Get first value of [array].

```javascript
const firstOf = require('@extra-array/first');
// firstOf(<array>, [index=0])

firstOf(['g', 'a', 'r', 'f', 'i', 'e', 'l', 'd']);
// 'g'
firstOf(['o', 'd', 'i', 'e'], 2);
// 'i'
```
