Get middle value of [array].

```javascript
const middleOf = require('@extra-array/middle');
// middleOf(<array>, [index=0])

middleOf(['m', 'a', 'r', 'a', 'm', 'e', 'l']);
// 'm'
middleOf(['m', 'i', 'l', 'k', 'm', 'a', 'n'], 1);
// 'i'
middleOf(['l', 'u', 'k', 'a'], -1);
// 'a'
```
